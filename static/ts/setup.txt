plugin.tx_keuserregister {

  # set user folder and default group
  userDataPID =
  defaultUsergroup =

  # set template file and css file
  templateFile = typo3conf/ext/ke_userregister/res/template/tx_keuserregister.tmpl
  cssFile = typo3conf/ext/ke_userregister/res/css/ke_userregister.css

  # use email as username?
  emailIsUsername = 1

  # password settings
  password {
    minLength = 6
    
    # possible values: none, md5, salted
    # use "salted" if you use the extension "saltedpasswords" with TYPO3 4.3
    encryption = salted
  }

  # email notification settings
  notification {
    from_email =
    from_name = TYPO3 User Registration
  }

  # how long are registration hash values accepted?
  hashDays = 7

  upload {
    maxFileSize = 5242880
    allowedFileTypes = jpg,gif,png
    path = uploads/tx_keuserregister/
  }

  # define the fields used in creation of new users
  create.fields {
    email {
      type = text
      eval = required,email
      tooltip =
    }
    username {
      type = text
      eval = required
      tooltip =
    }
    password {
      type = password
      eval = required
      tooltip =
    }
	# select: comma separated values are values of options. labels are set via locallang.
    gender {
      type = select
      values = 0,1
      tooltip =
    }
    first_name {
      type = text
      tooltip =
    }
    last_name {
      type = text
      tooltip =
    }
    # directmail: comma separated pids of pages with direct_mail-categories, leave emptye for "all categories"
    module_sys_dmail_category {
      type = directmail
      values =
      tooltip =
    }
    image {
      type = image
      file.maxW = 50
      file.maxH = 50
    }
  }

  # define the fields used in edit form
  edit.fields {
    email {
      type = text
      eval = required,email
    }
    username {
      type = text
      eval = required
    }
    gender {
      type = select
      values = 0,1
    }
    first_name {
      type = text
    }
    last_name {
      type = text
    }
    module_sys_dmail_category {
      type = directmail
      values =
    }
  }
}
