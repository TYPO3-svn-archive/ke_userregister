plugin.tx_keuserregister {

  # set user folder and default group
  userDataPID =
  defaultUsergroup =

  # set template file and css file
  templateFile = typo3conf/ext/ke_userregister/res/template/tx_keuserregister.tmpl
  cssFile = typo3conf/ext/ke_userregister/res/css/ke_userregister.css

  # use email as username?
  emailIsUsername = 1

  # password settings
  password {
    minLength = 6
    useMd5 = 1
  }

  # email notification settings
  notification {
    from_email = info@example.com
    from_name = USER REGISTRATION
  }

  # how long are registration hash values accepted?
  hashDays = 7

  upload {
    maxFileSize = 5242880
    allowedFileTypes = jpg,gif,png
  }

  # define the fields used in creation of new users
  create.fields {
    email {
      type = text
      eval = required,email
    }
    username {
      type = text
      eval = required
    }
    password {
      type = password
      eval = required
    }
	# select: comma separated values are values of options. labels are set via locallang.
    gender {
      type = select
      values = 0,1
    }
    first_name {
      type = text
	  tooltip = Please enter your first name.
    }
    last_name {
      type = text
	  tooltip = Please enter your last name.
    }
	# directmail: comma separated pids of pages with direct_mail-categories, leave emptye for "all categories"
    module_sys_dmail_category {
      type = directmail
      values =
	  tooltip = Register our newsletter.
    }
  }

  # define the fields used in edit form
  edit.fields {
    email {
      type = text
      eval = required,email
    }
    username {
      type = text
      eval = required
    }
    gender {
      type = select
      values = 0,1
    }
    first_name {
      type = text
    }
    last_name {
      type = text
    }
    module_sys_dmail_category {
      type = directmail
      values =
    }
  }
}
